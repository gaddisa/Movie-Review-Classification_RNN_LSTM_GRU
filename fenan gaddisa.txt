    <script>
        function updateCountdown() {
            // Get current time
            const now = new Date();
            
            // Set target time to 8:00 PM today
            const target = new Date();
            target.setHours(20, 0, 0, 0); // 8:00 PM
            
            // If it's already past 8:00 PM, set for next day
            if (now > target) {
                target.setDate(target.getDate() + 1);
            }
            
            // Calculate difference in seconds
            const diff = (target - now) / 1000;
            
            // Calculate hours, minutes, seconds
            const hours = Math.floor(diff / 3600);
            const minutes = Math.floor((diff % 3600) / 60);
            const seconds = Math.floor(diff % 60);
            
            // Format with leading zeros
            const formatted = `${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
            
            // Update countdown display
            document.getElementById('countdown-timer').textContent = formatted;
            
            // If countdown reaches zero, refresh the page
            if (diff <= 0) {
                window.location.reload();
            }
        }
        
        // Update countdown every second
        updateCountdown();
        setInterval(updateCountdown, 1000);
    </script>
