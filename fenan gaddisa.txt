    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const sssFileInput = document.getElementById('sss_file');
            const sssFileError = document.getElementById('sssFileError');
            const uploadSSSForm = document.getElementById('uploadSSSForm');

            sssFileInput.addEventListener('change', function(event) {
                const file = event.target.files[0];
                const allowedTypes = ['application/pdf'];
                const maxSize = 50 * 1024 * 1024; // 8MB in bytes

                if (file) {
                    if (!allowedTypes.includes(file.type)) {
                        sssFileError.textContent = 'Invalid file type. Please upload a PDF file.';
                        sssFileError.style.display = 'block';
                        sssFileInput.value = ''; // Clear the file input
                    } else if (file.size > maxSize) {
                        sssFileError.textContent = 'File size exceeds 8MB. Please upload a smaller file.';
                        sssFileError.style.display = 'block';
                        sssFileInput.value = ''; // Clear the file input
                    } else {
                        sssFileError.style.display = 'none'; // Hide error if valid
                    }
                }
            });

            uploadSSSForm.addEventListener('submit', function(event) {
                const file = sssFileInput.files[0];
                const maxSize = 50 * 1024 * 1024; // 8MB in bytes

                if (file && file.size > maxSize) {
                    event.preventDefault(); // Prevent form submission
                    sssFileError.textContent = 'File size exceeds 50MB. Please upload a smaller file.';
                    sssFileError.style.display = 'block';
                }
            });
        });
    </script>

    <script>
        document.addEventListener("DOMContentLoaded", function() {
            const completeTasks = document.querySelectorAll('.complete-task');
            const statusBadges = document.querySelectorAll('.status-badge');
            const submitButton = document.getElementById('submitButton');


            function checkCompletion() {
                let allCompleted = true;
                statusBadges.forEach(badge => {
                    if (badge.textContent.trim() === "Not Started") {
                        allCompleted = false;
                    }
                });

                submitButton.disabled = !allCompleted;
            }

            completeTasks.forEach(task => {
                task.addEventListener('click', function() {
                    const row = this.closest('tr');
                    const statusBadge = row.querySelector('.status-badge');
                    statusBadge.textContent = "Completed";
                    statusBadge.classList.remove('bg-warning', 'text-dark');
                    statusBadge.classList.add('bg-success', 'text-light');
                    checkCompletion();
                });
            });

        });
    </script>

    <script>
        document.addEventListener("DOMContentLoaded", function() {
            let allQuestionsAnswered = @json($dct_status === 'Completed'); // Ensure this is a boolean
            let sssExists = @json($sss_exists); // Ensure this is a boolean
            let serExists = @json($ser_exists); // Ensure this is a boolean
            let applicationStatus = @json($application->status); // Ensure this is a string
            let applicationId = @json($application->id);

            let submitButton = document.getElementById("submitButton");
            let submitParagraph = document.getElementById("submit-paragraph");
            let reuploadButtonSSS = document.getElementById("reuploadButtonSSS"); // Get Reupload button
            let reuploadButtonSER = document.getElementById("reuploadButtonSER"); // Get Reupload button


            // Debugging: Log the values
            console.log('allQuestionsAnswered:', allQuestionsAnswered);
            console.log('sssExists:', sssExists);
            console.log('serExists:', serExists);
            console.log('applicationStatus:', applicationStatus);

            // Activate button only when all conditions are met
            if (allQuestionsAnswered && sssExists && serExists && (applicationStatus === "profile_accepted" ||
                    applicationStatus === "desk_review_revision_requested")) {
                submitButton.removeAttribute("disabled");
                submitParagraph.innerHTML =
                    `<strong>All tasks completed!</strong> Click "Finish and Submit" to submit your documents.`;
            } else {
                console.log('Button disabled due to unmet conditions.');
            }

            // Submit function
            submitButton.addEventListener("click", function() {
                Swal.fire({
                    title: 'Are you sure?',
                    text: "This action cannot be undone. Are you sure you want to submit your application for assessment?",
                    icon: 'warning',
                    showCancelButton: true,
                    confirmButtonColor: '#3085d6',
                    cancelButtonColor: '#d33',
                    confirmButtonText: 'Yes, submit it!'
                }).then((result) => {
                    if (result.isConfirmed) {
                        submitButton.disabled = true;
                        submitButton.innerHTML =
                            '<i class="fas fa-spinner fa-spin"></i> Submitting...';

                        fetch("{{ route('backend.applications.updateApplicationStatus', ['application_id' => $application->id]) }}", {
                                method: "POST",
                                headers: {
                                    "X-CSRF-TOKEN": "{{ csrf_token() }}",
                                    "Content-Type": "application/json"
                                },
                                body: JSON.stringify({})
                            })
                            .then(response => response.json())
                            .then(data => {
                                if (data.success) {
                                    Swal.fire({
                                        icon: 'success',
                                        title: 'Success!',
                                        text: data.message,
                                        toast: true,
                                        position: 'top-end',
                                        showConfirmButton: false,
                                        timer: 3000
                                    });
                                    submitButton.style.display =
                                        "none"; // Hide button after submission
                                    submitParagraph.innerHTML =
                                        `<strong>Documents Submitted!</strong> Your application is now under review.`;
                                    if (reuploadButtonSSS) {
                                        reuploadButtonSSS.style.display = "none";
                                    }
                                    if (reuploadButtonSER) {
                                        reuploadButtonSER.style.display = "none";
                                    }
                                } else {
                                    Swal.fire({
                                        icon: 'error',
                                        title: 'Error!',
                                        text: data.message,
                                        toast: true,
                                        position: 'top-end',
                                        showConfirmButton: false,
                                        timer: 3000
                                    });
                                }
                            })
                            .catch(error => {
                                console.error("Error:", error);
                                Swal.fire("Error", "Something went wrong!", "error");
                                submitButton.disabled = false;
                                submitButton.innerHTML = originalButtonText;
                            });
                    }
                });
            });
        });
    </script>
