    <!-- SweetAlert Toast Script (from backend) -->
    <script>
        document.addEventListener('DOMContentLoaded', function() {
            window.addEventListener('showToast', function(event) {
                // Handle both array and object formats
                const detail = Array.isArray(event.detail) ? event.detail[0] : event.detail;
                const type = detail?.type || 'info';
                const message = detail?.message || 'Operation completed';

                // Ensure Swal is available
                if (typeof Swal !== 'undefined') {
                    Swal.fire({
                        toast: true,
                        position: 'top-end',
                        icon: type,
                        text: message,
                        showConfirmButton: false,
                        timer: 3000,
                        timerProgressBar: true,
                    });
                } else {
                    console.error('SweetAlert2 not loaded', {
                        type,
                        message
                    });
                    // Fallback to regular alert if SweetAlert isn't available
                    alert(`${type.toUpperCase()}: ${message}`);
                }
            });
        });
    </script>
