<?php

namespace Modules\Program\Http\Controllers\Backend;

use App\Authorizable;
use Illuminate\Support\Str;
use RealRashid\SweetAlert\Facades\Alert;
use App\Http\Controllers\Backend\BackendBaseController;
use Illuminate\Http\Request;

class ProgramsController extends BackendBaseController
{
    use Authorizable;

    public function __construct()
    {
        // Page Title
        $this->module_title = 'Programs';

        // module name
        $this->module_name = 'programs';

        // directory path of the module
        $this->module_path = 'program::backend';

        // module icon
        $this->module_icon = 'fa-regular fa-sun';

        // module model name, path
        $this->module_model = "Modules\Program\Models\Program";
    }

    public function index_data(?array $columns = null, array $additionalButtons = [], array $additionalData = [])
    {
        // Specify custom columns for the Programs module
        $columns = $columns ?? ['id', 'name', 'program_level', 'institute_type', 'accreditation_type', 'status', 'updated_at'];

        // Call the parent method with the specified columns and additional buttons
        return parent::index_data($columns, $additionalButtons);
    }

    public function store(Request $request)
    {
        $module_title = $this->module_title;
        $module_name = $this->module_name;
        $module_path = $this->module_path;
        $module_icon = $this->module_icon;
        $module_model = $this->module_model;
        $module_name_singular = Str::singular($module_name);

        $module_action = 'Store';

        // Validate the request
        $validatedData = $request->validate([
            'name' => 'required|string|max:255',
            'program_level' => 'required|string|max:255',
            'institute_type' => 'required|string|max:255',
            'accreditation_type' => 'required|string|in:Program Level,Institutional Accreditation',
            'status' => 'required|string',
        ]);

        // Create the description column value
        $description = sprintf(
            '%s %s Program (%s) - %s',
            $validatedData['name'],
            $validatedData['program_level'],
            $validatedData['institute_type'],
            $validatedData['accreditation_type']
        );

        // Merge the description into the validated data
        $validatedData['description'] = $description;

        // Save the program
        $$module_name_singular = $module_model::create($validatedData);

        // Notify the user
        Alert::toast(Str::singular($module_title) . "' Added Successfully", 'success')
            ->position('top-end')
            ->autoClose(5000)
            ->timerProgressBar();

        // Log the user action
        logUserAccess($module_title . ' ' . $module_action . ' | Id: ' . $$module_name_singular->id);

        // Redirect back to the program list
        return redirect("admin/{$module_name}");
    }
}
