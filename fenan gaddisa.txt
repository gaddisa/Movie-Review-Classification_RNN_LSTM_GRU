<?php

namespace App\Http\Controllers\Auth;

use App\Http\Controllers\Controller;
use Illuminate\Auth\Events\Verified;
use Illuminate\Http\RedirectResponse;
use App\Models\User;
use RealRashid\SweetAlert\Facades\Alert;

class VerifyEmailController extends Controller
{
    /**
     * Mark the user's email address as verified.
     */
    public function __invoke($id, $hash): RedirectResponse
    {
        $user = User::find($id);

        if (!$user) {
            Alert::error('Error', 'Invalid verification link. User not found.')->autoClose(5000);
            return redirect('/login');
        }

        // Verify the hash
        if (!hash_equals($hash, sha1($user->getEmailForVerification()))) {
            Alert::error('Error', 'Invalid verification link. Hash mismatch.')->autoClose(5000);
            return redirect('/login');
        }

        // Check if already verified
        if ($user->hasVerifiedEmail()) {
            Alert::info('Already Verified', 'Your email is already verified.')->autoClose(5000);
            return redirect('/login');
        }

        // Mark as verified
        $user->email_verified_at = now();
        $user->save();

        event(new Verified($user));

        Alert::success('Email Verified', 'Your email has been successfully verified. You can now login.')->autoClose(5000);
        return redirect('/login');
    }
}
