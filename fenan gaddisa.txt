<?php

namespace App\Models;

use Modules\DeskReview\Models\DeskReview;
use Illuminate\Database\Eloquent\Factories\HasFactory;

class Indicator extends BaseModel
{
    use HasFactory;

    protected $fillable = ['substandard_id', 'name', 'description', 'created_by', 'updated_by'];

    public function substandard()
    {
        return $this->belongsTo(Substandard::class);
    }
    
    // Define relationship with DeskReview
    public function deskReviews()
    {
        return $this->hasMany(DeskReview::class, 'indicator_id');
    }

    public function questions()
    {
        return $this->hasMany(Question::class);
    }

    // Many-to-many relationship with attachments through indicator_attachment pivot table
    public function attachments()
    {
        return $this->belongsToMany(
            \Modules\Attachement\Models\Attachement::class, // Related model
            'indicator_attachment', // Pivot table name
            'indicator_id', // Foreign key on the pivot table for this model
            'attachment_id' // Foreign key on the pivot table for the related model
        )->withTimestamps();
    }
}
