INSERT INTO model_has_roles (role_id, model_type, model_id)
SELECT r.id, 'App\\User', u.id
FROM users u
JOIN roles r ON r.name = 'super admin'
WHERE u.email = 'tigistkorsa22@gmail.com'
  AND NOT EXISTS (
    SELECT 1 FROM model_has_roles mr
    WHERE mr.role_id = r.id AND mr.model_type = 'App\\User' AND mr.model_id = u.id
  );
