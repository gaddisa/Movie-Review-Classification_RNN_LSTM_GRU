<script>
    // Update live clock
    function updateClock() {
        const now = new Date();
        const time = now.toLocaleTimeString();
        if (document.getElementById('liveClock')) {
            document.getElementById('liveClock').textContent = time;
        }
        setTimeout(updateClock, 1000);
    }

    document.addEventListener('DOMContentLoaded', function() {
        updateClock();

        // Add animation to notification badge if exists
        const notificationBadge = document.querySelector('.accreditation-badge-alert');
        if (notificationBadge) {
            notificationBadge.style.animation = 'pulse 2s infinite';
        }
    });

    // Pulse animation for alerts
    @keyframes pulse {
        0 % {
            transform: scale(1);
            box - shadow: 0 0 0 0 rgba(231, 76, 60, 0.7);
        }
        70 % {
            transform: scale(1.1);
            box - shadow: 0 0 0 5 px rgba(231, 76, 60, 0);
        }
        100 % {
            transform: scale(1);
            box - shadow: 0 0 0 0 rgba(231, 76, 60, 0);
        }
    }
</script>
